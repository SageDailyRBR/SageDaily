        // Gestione lingua

        let currentLang = localStorage.getItem('app_language') || 'it';

 

        function getTodayString() {

            return new Date().toISOString().split('T')[0];

        }

 

        function getQuoteForToday() {

            const today = getTodayString();

            const key = `quote_${currentLang}_${today}`;

            const saved = localStorage.getItem(key);

           

            if (saved) return saved;

           

            const quotes = quotesDB[currentLang];

            const quote = quotes[Math.floor(Math.random() * quotes.length)];

            localStorage.setItem(key, quote);

           

            // Cleanup vecchie frasi

            cleanupOldQuotes();

            return quote;

        }

 

        function cleanupOldQuotes() {

            const weekAgo = new Date();

            weekAgo.setDate(weekAgo.getDate() - 7);

            const weekAgoStr = weekAgo.toISOString().split('T')[0];

           

            for (let i = 0; i < localStorage.length; i++) {

                const key = localStorage.key(i);

                if (key && key.startsWith('quote_')) {

                    const datePart = key.split('_').pop();

                    if (datePart && datePart < weekAgoStr) {

                        localStorage.removeItem(key);

                    }

                }

            }

        }

 

        function formatDate() {

            const options = { weekday: 'long', day: 'numeric', month: 'long' };

            const locale = currentLang === 'it' ? 'it-IT' : 'en-US';

            return translations[currentLang].todayIs + new Date().toLocaleDateString(locale, options);

        }

        function updateLanguageUI() {

            document.querySelectorAll('.language-option').forEach(el => {

                el.classList.remove('selected');

            });

            document.getElementById(`lang-${currentLang}`).classList.add('selected');

            document.getElementById('settingsTitle').textContent = translations[currentLang].settings;

        }

        function startAnimation() {

            const t = translations[currentLang];

            document.getElementById('loadingText').textContent = t.loading;

            document.getElementById('closeHint').textContent = t.closeHint;

            document.getElementById('closingText').innerHTML = t.closing;
